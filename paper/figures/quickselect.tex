datatype list = Empty 
              | Cons of int * list


fun part (p, Empty) =(0,Empty,Empty) 
  | part (p, Cons (h,t)) = 
    let val (n,le,ri) = part (p,t) 
    in if h < p 
       then (n+1,Cons(h,le),ri) 
       else (n,le,Cons(h,ri))


qsel: list * int -> int
fun qsel (Empty, k) = 0
  | qsel (Cons ht, k) =
    let val (i,le,ri) = part ht in 
    case compare k i of
      LT => qsel (le, k)
    | EQ => #1 ht
    | GT => qsel (ri, k-i-1)
